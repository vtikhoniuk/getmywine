async def find_wines_for_suggestion(
    suggestion: WineSuggestion,
    wine_repo: WineRepository,
    user_profile: Optional[UserProfile] = None,
    limit: int = 1
) -> list[Wine]:
    """–ù–∞—Ö–æ–¥–∏—Ç –≤–∏–Ω–∞ –∏–∑ –∫–∞—Ç–∞–ª–æ–≥–∞ –ø–æ–¥ –∫–æ–Ω–∫—Ä–µ—Ç–Ω–æ–µ –ø—Ä–µ–¥–ª–æ–∂–µ–Ω–∏–µ."""

    query = suggestion.wine_query

    # –ë–∞–∑–æ–≤—ã–µ —Ñ–∏–ª—å—Ç—Ä—ã
    filters = {}
    if "wine_type" in query:
        filters["wine_type"] = query["wine_type"]
    if "body_min" in query:
        filters["body_min"] = query["body_min"]
    if "body_max" in query:
        filters["body_max"] = query["body_max"]
    if "price_min" in query:
        filters["price_min"] = query["price_min"]
    if "price_max" in query:
        filters["price_max"] = query["price_max"]

    # –ü–µ—Ä—Å–æ–Ω–∞–ª–∏–∑–∞—Ü–∏—è
    if user_profile and query.get("personalized"):
        if user_profile.budget_range:
            filters["price_max"] = user_profile.budget_range.max
        if user_profile.preferred_sweetness:
            filters["sweetness"] = user_profile.preferred_sweetness

    # –ü–æ–∏—Å–∫ "–æ—Ç–∫—Ä—ã—Ç–∏–π" ‚Äî —Ä–µ–≥–∏–æ–Ω—ã, –∫–æ—Ç–æ—Ä—ã–µ –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—å –Ω–µ –ø—Ä–æ–±–æ–≤–∞–ª
    if query.get("discovery_region") and user_profile:
        tried_countries = user_profile.tried_countries
        # –ò—Å–∫–ª—é—á–∞–µ–º —É–∂–µ –ø–æ–ø—Ä–æ–±–æ–≤–∞–Ω–Ω—ã–µ
        filters["exclude_countries"] = tried_countries

    wines = await wine_repo.get_list(limit=limit, **filters)

    # –ï—Å–ª–∏ –Ω–∏—á–µ–≥–æ –Ω–µ –Ω–∞–π–¥–µ–Ω–æ ‚Äî —Ä–∞—Å—à–∏—Ä—è–µ–º –∫—Ä–∏—Ç–µ—Ä–∏–∏
    if not wines:
        wines = await wine_repo.get_list(limit=limit)

    return wines
```

def generate_explanation(
    wine: Wine,
    suggestion: WineSuggestion,
    user_profile: Optional[UserProfile] = None
) -> str:
    """–ì–µ–Ω–µ—Ä–∏—Ä—É–µ—Ç –ø–µ—Ä—Å–æ–Ω–∞–ª–∏–∑–∏—Ä–æ–≤–∞–Ω–Ω–æ–µ –æ–±—ä—è—Å–Ω–µ–Ω–∏–µ –≤—ã–±–æ—Ä–∞."""

    parts = []

    # –°–≤—è–∑—å —Å –∫–æ–Ω—Ç–µ–∫—Å—Ç–æ–º (—Å–µ–∑–æ–Ω, –ø—Ä–∞–∑–¥–Ω–∏–∫)
    if suggestion.angle == "–°–µ–∑–æ–Ω–Ω–æ–µ":
        if wine.body >= 4:
            parts.append("–ù–∞—Å—ã—â–µ–Ω–Ω–æ–µ –∏ —Å–æ–≥—Ä–µ–≤–∞—é—â–µ–µ")
        else:
            parts.append("–û—Å–≤–µ–∂–∞—é—â–µ–µ –∏ –ª—ë–≥–∫–æ–µ")

    # –°–≤—è–∑—å —Å –ø—Ä–æ—Ñ–∏–ª–µ–º
    if user_profile:
        if wine.sweetness == user_profile.preferred_sweetness:
            parts.append(f"–∏–º–µ–Ω–Ω–æ —Ç–∞ —Å–ª–∞–¥–æ—Å—Ç—å, –∫–æ—Ç–æ—Ä—É—é –≤—ã –ª—é–±–∏—Ç–µ")
        if wine.country in user_profile.favorite_countries:
            parts.append(f"–∏–∑ –ª—é–±–∏–º–æ–≥–æ –≤–∞–º–∏ {wine.country}")

    # –í–∫—É—Å–æ–≤—ã–µ –Ω–æ—Ç—ã
    if wine.tasting_notes:
        parts.append(f"—Å –Ω–æ—Ç–∞–º–∏ {wine.tasting_notes}")

    # –ì–∞—Å—Ç—Ä–æ–Ω–æ–º–∏—è
    if wine.food_pairings:
        parts.append(f"–æ—Ç–ª–∏—á–Ω–æ —Å {', '.join(wine.food_pairings[:2])}")

    return ". ".join(parts) + "."
```
